## 默认按钮

```jsx
import React from "react";
import { Button } from "vmdc-ui";

export default () => <Button>默认按钮</Button>
```

## 实心按钮

```jsx
import React from "react";
import { Button } from "vmdc-ui";

export default () => <Button type="contained">实心按钮</Button>
```
